integer function fi(integer x, integer y)
{
declare integer z;
z= x*(x+y);
return z;
};
string function fs (string a, string b)
{
declare string c;
c=a+b;
return c;
};
main
{
declare integer x;
declare integer y;
declare integer z;
declare string sa;
declare string sb;
declare string sc;
x = 1;
y = 5;
sa = '1234567890';
sb = '1234567890';
z = fi(x,y);
sc = fs(sa,sb);
print 'контрольный пример';
print z;
print sc;
return 0;
};
